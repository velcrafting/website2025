export function normalizeScore(x: number, min = 0, max = 100) {
  if (max === min) return 0;
  const clamped = Math.min(Math.max(x, min), max);
  return (clamped - min) / (max - min);
}

## 3) Image sizes that “just work”
You’ve set `Figure` to serve `sizes="(min-width: 1024px) 860px, 100vw"`. To be crisp on modern screens, provide at least 2× that width for raster assets.

Recommended source asset sizes:
- OG image: exactly 1200×630. PNG or JPEG. Keep under ~600 KB.
- Article hero: 1720–2000 px wide. 16:9 or 3:2. JPEG for photos, PNG for UI.
- Inline figure: 1400–1800 px wide. Match the content aspect ratio. PNG for charts/UI; JPEG for photos.
- Gallery tiles: 800–1200 px on the long edge.
- Avatar: 640×640.
- Logos/icons: SVG whenever possible.

Format guidance:
- Photographic: JPEG (quality ~75–82).
- UI, diagrams, sharp edges: PNG or SVG.
- Never upscale tiny sources. Compress with `squoosh.app` or `sharp`.

If you want more precise art direction, bump the `sizes` in `Figure`:
